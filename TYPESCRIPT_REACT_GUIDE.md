## 1. Project Structure

```
pivotx/
├── src/
│   ├── core/                    ← Pure TypeScript — no framework dependency
│   │   ├── types.ts             ← IPoint, IDrawable, IShape, LoopCallback
│   │   ├── Point.ts             ← Point(x, y) factory
│   │   ├── Canvas.ts            ← Canvas class with game loop
│   │   ├── shapes/
│   │   │   ├── Circle.ts
│   │   │   ├── Rectangle.ts
│   │   │   ├── Line.ts
│   │   │   └── Label.ts
│   │   └── index.ts             ← re-exports everything from core
│   │
│   └── react/                   ← React wrapper — thin layer over core
│       ├── PivotCanvas.tsx      ← <PivotCanvas> + CanvasContext
│       ├── hooks/
│       │   └── useGameLoop.ts   ← useGameLoop(callback)
│       ├── components/
│       │   └── shapes.tsx       ← <PivotCircle>, <PivotRect>, <PivotLine>, <PivotLabel>
│       └── index.ts
│
├── dist/                        ← generated by `npm run build`
│   ├── pivotx.umd.js            ← browser drop-in  (window.PivotX)
│   ├── pivotx.esm.js            ← ES module        (import { Canvas } from '@colon-dev/pivotx')
│   ├── pivotx.cjs.js            ← CommonJS          (require('pivotx'))
│   ├── react.esm.js             ← React ESM
│   ├── react.cjs.js             ← React CJS
│   └── index.d.ts               ← TypeScript types for everything
│
├── examples/
│   ├── vanilla.html             ← Vanilla JS + UMD bundle example
│   ├── typescript-example.ts   ← TypeScript import example
│   └── react-example.tsx       ← All three React patterns
│
├── rollup.config.js
├── tsconfig.json
└── package.json
```

**Key design principle:** The `core/` layer has zero framework dependencies. React is an optional thin wrapper around it. If you stop using React, your game logic doesn't change.

## 2. Installation & Build

### If you're publishing to npm

```bash
# Install dev dependencies
npm install

# Build all outputs (UMD + ESM + CJS + types)
npm run build

# Watch mode during development
npm run dev
```

This produces five files in `dist/` covering every use case.

### If you're using it locally (no npm publish)

```bash
# In your game project
npm install ../path/to/@colon-dev/pivotx

# Or copy dist/ files directly for Vanilla JS use
```

## 3. The Three Usage Modes

| Mode | File to use | Import style | Requires build step? |
|---|---|---|---|
| Vanilla JS | `dist/pivotx.umd.js` | `<script src="...">` → `window.PivotX` | No |
| TypeScript | `dist/pivotx.esm.js` | `import { Canvas } from '@colon-dev/pivotx'` | Yes (your project) |
| React | `dist/react.esm.js` | `import { PivotCanvas } from '@colon-dev/pivotx/react'` | Yes (your project) |

## 4. Vanilla JS (UMD — no build step)

Drop the UMD bundle into a `<script>` tag. Everything is on `window.PivotX`.

```html
<script src="dist/pivotx.umd.js"></script>
<canvas id="game" width="600" height="400"></canvas>

<script>
  var Canvas    = PivotX.Canvas;
  var Circle    = PivotX.Circle;
  var Rectangle = PivotX.Rectangle;
  var Line      = PivotX.Line;
  var Label     = PivotX.Label;
  var Point     = PivotX.Point;

  var canvas = new Canvas("game");
  var W = canvas.getWidth();
  var H = canvas.getHeight();

  var ball = { x: W/2, y: H/2, r: 24, vx: 200, vy: 150 };

  canvas.startLoop(function(dt) {
    canvas.clear();

    ball.x += ball.vx * dt;
    ball.y += ball.vy * dt;
    if (ball.x < ball.r || ball.x > W - ball.r) ball.vx *= -1;
    if (ball.y < ball.r || ball.y > H - ball.r) ball.vy *= -1;

    var c = new Circle(Point(ball.x, ball.y), ball.r);
    c.fillColor = "tomato";
    canvas.add(c);
  });
</script>
```

This is identical to your original `pIvotX.js` API — fully backward compatible.

## 5. TypeScript (ESM)

Import with full type safety. TypeScript will catch wrong types at compile time.

```ts
import { Canvas, Circle, Rectangle, Line, Label, Point } from '@colon-dev/pivotx';
import type { IPoint, LoopCallback } from '@colon-dev/pivotx';

const canvas = new Canvas('game');
const W      = canvas.getWidth();
const H      = canvas.getHeight();

// TypeScript knows IPoint is { x: number, y: number }
const start: IPoint = Point(0, 0);
const end:   IPoint = Point(W, H);

const line       = new Line(start, end);
line.strokeColor = 'crimson';   // ✅ string — OK
line.strokeColor = 42;          // ❌ compile error: not a string
line.lineWidth   = 3;           // ✅ number — OK
canvas.add(line);

// The game loop callback is typed as (dt: number) => void
const loop: LoopCallback = (dt) => {
  canvas.clear();
  // dt is inferred as number — no annotation needed
};
canvas.startLoop(loop);
```

### Type-safe custom game state

```ts
interface Player {
  pos:   IPoint;
  size:  number;
  speed: number;
  color: string;
}

const player: Player = {
  pos:   Point(300, 200),
  size:  30,
  speed: 250,
  color: '#e94560',
};
```

## 6. React

Import shape components and wrap them in `<PivotCanvas>`.

```tsx
import { PivotCanvas, PivotCircle, PivotRectangle, PivotLabel } from '@colon-dev/pivotx/react';

function MyScene() {
  return (
    <PivotCanvas width={600} height={400} background="#1a1a2e">
      <PivotCircle
        center={{ x: 300, y: 200 }}
        radius={60}
        fill="tomato"
        stroke="white"
        lineWidth={3}
      />
      <PivotLabel
        text="Hello pIvotX"
        position={{ x: 300, y: 360 }}
        font="20px Arial"
        fill="white"
      />
    </PivotCanvas>
  );
}
```

## 7. useGameLoop Hook

`useGameLoop(callback)` starts a `requestAnimationFrame` loop that lives as long as the component is mounted, and automatically stops when the component unmounts.

```tsx
import { useGameLoop } from '@colon-dev/pivotx/react';

function MyGame() {
  useGameLoop((dt) => {
    // dt = seconds since last frame (~0.016 at 60fps)
    // update your game state here
    // React re-renders handle drawing
  });

  return <PivotCanvas>...</PivotCanvas>;
}
```

**Important:** The callback is stored in a ref internally, so it always sees the latest state and props without needing to be in the dependency array. You never need `useCallback` around it.

### Triggering re-renders from the loop

Shape components only re-draw when React re-renders. To drive animation, force a render each frame:

```tsx
import { useState, useRef } from 'react';
import { useGameLoop } from '@colon-dev/pivotx/react';

function AnimatedBall() {
  const pos = useRef({ x: 300, y: 200, vx: 200, vy: 150 });
  const [, tick] = useState(0);  // just a counter to force renders

  useGameLoop((dt) => {
    pos.current.x += pos.current.vx * dt;
    pos.current.y += pos.current.vy * dt;
    tick(n => n + 1);  // triggers re-render
  });

  return (
    <PivotCanvas width={600} height={400}>
      <PivotCircle center={pos.current} radius={24} fill="tomato" />
    </PivotCanvas>
  );
}
```

Use `useRef` for mutable game state (position, velocity, score) and `useState` only to trigger re-renders. This avoids unnecessary re-renders from state changes that don't affect the scene.

## 8. React Patterns: Static vs Animated

### Pattern A — Static (props drive everything)
Good for scenes that only change when React state changes (menus, scoreboards, UI).

```tsx
function Scoreboard({ score, lives }: { score: number; lives: number }) {
  return (
    <PivotCanvas width={300} height={80} background="#0f3460">
      <PivotLabel text={`Score: ${score}`}  position={{ x: 20, y: 25 }} fill="white" textAlign="left" />
      <PivotLabel text={`Lives: ${lives}`}  position={{ x: 20, y: 55 }} fill="#e94560" textAlign="left" />
    </PivotCanvas>
  );
}
```

### Pattern B — Animated loop (useRef + useGameLoop)
Good for anything that moves every frame.

```tsx
function BouncingBall() {
  const b = useRef({ x: 300, y: 200, vx: 180, vy: 120 });
  const [, tick] = useState(0);

  useGameLoop((dt) => {
    b.current.x += b.current.vx * dt;
    b.current.y += b.current.vy * dt;
    if (b.current.x < 20 || b.current.x > 580) b.current.vx *= -1;
    if (b.current.y < 20 || b.current.y > 380) b.current.vy *= -1;
    tick(n => n + 1);
  });

  return (
    <PivotCanvas width={600} height={400} background="#1a1a2e">
      <PivotCircle center={b.current} radius={20} fill="#e94560" />
    </PivotCanvas>
  );
}
```

### Pattern C — PivotCanvasHandle ref (raw ctx access)
For advanced rendering that doesn't fit the component model.

```tsx
import { useRef, useEffect } from 'react';
import { PivotCanvas, PivotCanvasHandle } from '@colon-dev/pivotx/react';

function AdvancedGame() {
  const canvasRef = useRef<PivotCanvasHandle>(null);

  useEffect(() => {
    const handle = canvasRef.current;
    if (!handle?.ctx) return;

    // Draw directly with the raw 2D context
    const ctx = handle.ctx;
    ctx.fillStyle = 'gold';
    ctx.fillRect(100, 100, 200, 50);
  }, []);

  return <PivotCanvas ref={canvasRef} width={600} height={400} />;
}
```

## 9. Custom Shapes

Any object that implements `IDrawable` works with `canvas.add()`.

```ts
import type { IDrawable } from '@colon-dev/pivotx';

// A custom star shape
class Star implements IDrawable {
  readonly tag = 'star';

  constructor(
    public cx:     number,
    public cy:     number,
    public points: number,
    public outer:  number,
    public inner:  number,
    public color:  string = 'gold',
  ) {}

  draw(ctx: CanvasRenderingContext2D): void {
    const step = Math.PI / this.points;
    ctx.beginPath();
    for (let i = 0; i < 2 * this.points; i++) {
      const r   = i % 2 === 0 ? this.outer : this.inner;
      const ang = i * step - Math.PI / 2;
      const x   = this.cx + Math.cos(ang) * r;
      const y   = this.cy + Math.sin(ang) * r;
      i === 0 ? ctx.moveTo(x, y) : ctx.lineTo(x, y);
    }
    ctx.closePath();
    ctx.fillStyle = this.color;
    ctx.fill();
  }
}

// Use it exactly like built-in shapes
const star = new Star(300, 200, 5, 60, 25, '#FFD700');
canvas.add(star);
```

The same class works in React too — call `shape.draw(ctx)` directly from a `useEffect` using the `PivotCanvasHandle` ref.

## 10. TypeScript Type Reference

```ts
// A plain coordinate object
interface IPoint {
  x: number;
  y: number;
}

// Anything that can be drawn — implement this for custom shapes
interface IDrawable {
  readonly tag: string;
  draw(ctx: CanvasRenderingContext2D): void;
}

// IDrawable + fill/stroke styling
interface IShape extends IDrawable {
  fillColor:   string | null;
  strokeColor: string | null;
  lineWidth:   number;
}

// The game loop callback signature
type LoopCallback = (dt: number) => void;

// Colour — any valid CSS colour string
type CSSColor = string;
// e.g. "red", "#ff0000", "rgb(255,0,0)", "rgba(255,0,0,0.5)", "hsl(0,100%,50%)"
```
